SELECT E.COD_PRT, E.NOME_PAC,E.NASCIMENTO, E.SEXO, A.COD_PROC, D.NOME_PRO,
C.DESCRICAO, A.DATA_REAL, A.TECNICA_UTILIZADA
FROM FAPRCCAD A, FAPACCAD B, FAPRDCAD C, FAPROCAD D, FAPRTCAD E
WHERE A.COD_PAC = B.COD_PAC
AND B.COD_PRT = E.COD_PRT
AND A.COD_PRO = D.COD_PRO
AND A.COD_PROC = C.COD_AMB
AND A.DATA_REAL between '01/07/2009' and '03/09/2010'
AND A.TECNICA_UTILIZADA LIKE 'V'
ORDER BY DATA_REAL
