          SELECT TI_CTR_CONTASPAGAR.SEQUENCIAL
                ,INTP_CTR_PROC.SEQUENCIAL AS SEQUENCIALCTR
                ,TI_CTR_CONTASPAGAR.CO_FORNECEDOR
                ,TI_CTR_CONTASPAGAR.NU_NOTA_FISCAL
                ,TI_CTR_CONTASPAGAR.NU_LANCAMENTO
                ,TI_CTR_CONTASPAGAR.CO_UNIDADE
               ,TI_CTR_CONTASPAGAR.DH_EMISSAO
                ,TI_CTR_CONTASPAGAR.NU_SERIE
                ,TI_CTR_CONTASPAGAR.CO_FIRMA
                ,TI_CTR_CONTASPAGAR.TIPO_OPER
            FROM INTP_CTR_PROC
                ,TI_CTR_CONTASPAGAR
                ,INTP_TABELAS_CTR
                ,INTP_EMP_INT_LAY
           WHERE 1=1 
             and INTP_CTR_PROC.ID_REG_REPOSITORIO IS NULL            
             and INTP_CTR_PROC.DATA_PROC          IS NULL
             AND INTP_CTR_PROC.SEQ_TI             = TI_CTR_CONTASPAGAR.SEQUENCIAL
             AND INTP_CTR_PROC.SEQ_TABELAS_CTR    = INTP_TABELAS_CTR.SEQUENCIAL
             AND INTP_TABELAS_CTR.COD_TABELA      = 'TI_CTR_CONTASPAGAR'
             AND TI_CTR_CONTASPAGAR.TIPO_CHAVE    = 'N'
             AND INTP_CTR_PROC.SEQ_EIL            = INTP_EMP_INT_LAY.SEQUENCIAL
             AND INTP_EMP_INT_LAY.COD_EIL         = '3060'
           ORDER BY TI_CTR_CONTASPAGAR.SEQUENCIAL;
