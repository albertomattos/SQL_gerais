SELECT B.COD_PAC, C.NOME_PAC, B.DATA_ENT, B.DATA_ALTA, A.DEP_RET, A.DOCUMENTO, A.COD_FORMA_PAGTO, A.DATA_MOV, A.NOM_RESP_PAGTO, A.REQ_DEV, A.NUM_BANCO, A.VALOR, A.NUM_CHEQUE, A.DSC_OBS, A.NOM_CIDADE
FROM FAMCACAD A, FAPACCAD B, FAPRTCAD C
WHERE A.COD_PAC = B.COD_PAC
AND B.COD_PRT = C.COD_PRT
AND B.COD_UNI IN ('0001')
AND A.DSC_OBS LIKE '%MED%'
AND A.DATA_MOV BETWEEN '01/09/2010' AND '13/09/2010'
